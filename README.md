# This project requires that you first install Node.js and npm

# Installing project dependencies
    1.  cd backend
    2.  npm install
    3.  cd ../frontend
    4.  npm install
    
# Running the application - localhost:3000
    1.  cd frontend
    2.  npm start

# Opening Sanity workbench - localhost:3333
    1.  cd backend
    2.  npm run dev
